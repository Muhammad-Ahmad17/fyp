\section{Your Role}
%===============================================================================

\begin{scopebox}
\textbf{You own:} Full MQTT pipeline (publisher $\rightarrow$ broker $\rightarrow$ bridge $\rightarrow$ consumers), Jetson deploy, 5 FastAPI services, PostgreSQL/MinIO/Redis, LangGraph agent, synthetic data, React dashboard, Docker/CI/CD/Grafana, event simulator.

\section{Sprint 1 (Weeks 1--4): Foundation \& Transport}
%===============================================================================

\begin{sprintbox}{{Your focus --- Repo; contract; MQTT; Redis}}

\begin{taskbox}
\textbf{Week 1--2}
\begin{itemize}
    \item Initialize \texttt{smart-surveillance/} monorepo; directory structure per architecture plan
    \item \texttt{infra/docker-compose.yml} skeleton; \texttt{.env.example}
    \item Draft \texttt{contracts/events.py} (Pydantic); share with Member B
    \item Dev environment: Python 3.11+, Node 20+, Docker Compose v2
    \item GitHub: branch rules, first PR, Jira board linked; create \texttt{sprint-1} tickets
\end{itemize}
\end{taskbox}

\begin{taskbox}
\textbf{Week 3--4}
\begin{itemize}
    \item Mosquitto (:1883) + Redis (:6379) in Compose
    \item \texttt{transport/mqtt\_redis\_bridge/} --- subscribe \texttt{ppe/\#}, XADD to streams
    \item \texttt{transport/event\_simulator/} --- contract-valid test events
    \item Streams: \texttt{ppe:alerts}, \texttt{ppe:metrics}, \texttt{ppe:heartbeat}
    \item Document MQTT topic map in \texttt{contracts/README.md}
\end{itemize}
\end{taskbox}

\begin{deliverablebox}
\textbf{Your Sprint 1 done when:}
\begin{itemize}
    \item \texttt{docker compose up} starts Mosquitto + Redis + bridge + simulator
    \item Events visible via \texttt{XRANGE ppe:alerts - + COUNT 5}
    \item Prerequisites checklist signed (\texttt{.prerequisite/MEMBER\_A\_PREREQUISITES.md})
\end{itemize}
\end{deliverablebox}

\begin{collabbox}
\textbf{Week 4 with Member B:} Sign frozen contract (\textbf{M0}). Review B's handoff dict draft; confirm your publisher can serialize it.
\end{collabbox}

\end{sprintbox}
