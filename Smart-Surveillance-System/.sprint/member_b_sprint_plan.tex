\section{Your Role}
%===============================================================================

\begin{scopebox}
\textbf{You own:} Three-track CV pipeline --- Worker Detection, PPE (helmet/vest), Hazard (fire/smoke/spill) --- plus compliance engine, evidence capture, and a stable \textbf{handoff dict} for Member A. All work in \texttt{edge/inference/} and \texttt{edge/compliance\_engine/}.

\section{Sprint 1 (Weeks 1--4): Dataset \& Handoff Spec}
%===============================================================================

\begin{sprintbox}{Your focus --- Data pipeline and integration boundary}

\begin{taskbox}
\textbf{Week 1--2}
\begin{itemize}
    \item Python venv; install OpenCV, Ultralytics, PyTorch; verify GPU/CUDA
    \item Clone repo; branch \texttt{feature/b-edge-readme}; first PR
    \item Source PPE/hazard datasets (Roboflow, Kaggle, custom capture)
    \item Set up LabelImg / Roboflow / CVAT; YOLO label format
    \item Create \texttt{edge/inference/README.md} --- 6 classes, three tracks diagram
\end{itemize}
\end{taskbox}

\begin{taskbox}
\textbf{Week 3--4}
\begin{itemize}
    \item Annotate first 200+ images (all 6 classes represented)
    \item Draft \textbf{CV handoff interface} in \texttt{edge/README.md}:
    \begin{verbatim}
    get_compliance_event(frame) -> dict
    # event, zone, worker_id, compliance_score,
    # confidence, detections[], image_path
    \end{verbatim}
    \item Run YOLO pretrained inference demo; save annotated sample outputs
    \item Sign prerequisites checklist
\end{itemize}
\end{taskbox}

\begin{deliverablebox}
\textbf{Your Sprint 1 done when:} Handoff spec reviewed by Member A; 200+ labels committed; pretrained demo runs locally.
\end{deliverablebox}

\begin{collabbox}
\textbf{Week 4 with Member A:} Align handoff dict fields with \texttt{contracts/events.py} (\textbf{M0}). You do not attend MQTT setup --- only review contract mapping document.
\end{collabbox}

\end{sprintbox}

\section{Sprint 2 (Weeks 5--8): Training \& Compliance Prototype}
%===============================================================================

\begin{sprintbox}{Your focus --- YOLO training + IoU + compliance logic}

\begin{taskbox}
\textbf{Week 5--6}
\begin{itemize}
    \item Expand dataset to 500+ images; \texttt{data.yaml} for Ultralytics
    \item Train YOLOv8n; monitor mAP on val set
    \item Implement IoU function: associate helmet/vest boxes to nearest worker
    \item Unit tests for IoU and debounce logic
\end{itemize}
\end{taskbox}

\begin{taskbox}
\textbf{Week 7--8}
\begin{itemize}
    \item Compliance engine prototype in \texttt{edge/compliance\_engine/}
    \item Per-zone rules: helmet + vest required (config file, not DB --- A owns DB)
    \item Rolling 5-frame compliance score
    \item Evidence: \texttt{cv2.imwrite} on violation; return \texttt{image\_path} in handoff dict
    \item Violation debounce (60s window per worker/event)
\end{itemize}
\end{taskbox}

\begin{deliverablebox}
\textbf{Your Sprint 2 done when:} Local demo: webcam $\rightarrow$ inference $\rightarrow$ handoff dict printed to console; mAP trending $\geq$ 0.60 on val.
\end{deliverablebox}

\begin{collabbox}
\textbf{Week 8:} Demo handoff JSON file to Member A for their test harness (\textbf{M1 prep}). Answer questions in integration session --- no Jetson required yet.
\end{collabbox}

\end{sprintbox}
